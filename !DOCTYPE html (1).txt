const chatContainer = document.getElementById('chat-container');

let i = 0;

function showMessage() {
    if(i >= messages.length) return;
    const msg = messages[i];
    const div = document.createElement('div');
    div.classList.add('message');
    if(msg.type === "admin") div.classList.add('admin');
    else if(msg.type === "Lan") div.classList.add('Lan');
    else if(msg.type === "Minh") div.classList.add('Minh');
    else div.classList.add('other');
    div.innerHTML = `<span class="sender">${msg.sender}</span>${msg.text.replace(/\n/g,'<br>')}`;
    chatContainer.appendChild(div);
    chatContainer.scrollTop = chatContainer.scrollHeight;
    i++;
    setTimeout(showMessage, 1000); // 1 giây một tin nhắn
}

showMessage();
